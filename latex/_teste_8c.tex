\begin{DoxyParams}{Parameters}
{\em pdt} & Nome do produto ou item comprado. \\
\hline
{\em cate\+\_\+g} & Categoria do gasto (1 para Roupas, 2 para Cosmeticos). \\
\hline
{\em preco} & Preco unitario do produto (deve ser maior ou igual a zero). \\
\hline
{\em qtde\+\_\+g} & Quantidade comprada (deve ser um valor inteiro maior que zero). \\
\hline
\end{DoxyParams}
\begin{DoxyReturn}{Returns}

\end{DoxyReturn}
\begin{DoxyNote}{Note}

\begin{DoxyItemize}
\item O parametro {\ttfamily cate\+\_\+g} aceita apenas os valores 1 (Roupa) ou 2 (Cosmetico).
\item O parametro {\ttfamily preco} aceita numeros decimais e inteiros positivos ou zero. Valores negativos tornam o gasto invalido.
\item O parametro {\ttfamily qtde\+\_\+g} aceita apenas inteiros maiores que zero. Valores menores ou iguais a zero tornam o gasto invalido. 
\end{DoxyItemize}
